### JavaScript da Landing Page
```javascript
// Controle do showcase de produtos no hero
class ScanologyLanding {
    constructor() {
        this.currentProduct = 0;
        this.products = ['simscan-e', 'kscan-magic', 'am-cell'];
        this.init();
    }
    
    init() {
        this.bindShowcaseControls();
        this.startAutoRotation();
        this.bindSmoothScroll();
        this.addScrollAnimations();
    }
    
    bindShowcaseControls() {
        document.querySelectorAll('.control-btn').forEach((btn, index) => {
            btn.addEventListener('click', () => {
                this.switchProduct(index);
            });
        });
    }
    
    switchProduct(index) {
        // Remove active classes
        document.querySelectorAll('.showcase-item').forEach(item => {
            item.classList.remove('active');
        });
        document.querySelectorAll('.control-btn').forEach(btn => {
            btn.classList.remove('active');
        });
        
        // Add active to selected
        document.querySelector(`[data-product="${this.products[index]}"]`).classList.add('active');
        document.querySelectorAll('.control-btn')[index].classList.add('active');
        
        this.currentProduct = index;
    }
    
    startAutoRotation() {
        setInterval(() => {
            this.currentProduct = (this.currentProduct + 1) % this.products.length;
            this.switchProduct(this.currentProduct);
        }, 5000); // 5 segundos
    }
    
    bindSmoothScroll() {
        document.querySelectorAll('a[href^="#"]').forEach(anchor => {
            anchor.addEventListener('click', (e) => {
                e.preventDefault();
                const target = document.querySelector(anchor.getAttribute('href'));
                if (target) {
                    target.scrollIntoView({
                        behavior: 'smooth',
                        block: 'start'
                    });
                }
            });
        });
    }
    
    addScrollAnimations() {
        const observerOptions = {
            threshold: 0.1,
            rootMargin: '0px 0px -50px 0px'
        };
        
        const observer = new IntersectionObserver((entries) => {
            entries.forEach(entry => {
                if (entry.isIntersecting) {
                    entry.target.classList.add('animate-in');
                }
            });
        }, observerOptions);
        
        // Observar elementos para animação
        document.querySelectorAll('.category-card, .feature-item, .app-card').forEach(el => {
            observer.observe(el);
        });
    }
}

// Inicializar quando DOM estiver pronto
document.addEventListener('DOMContentLoaded', () => {
    new ScanologyLanding();
});

// CSS para animações
const style = document.createElement('style');
style.textContent = `
    .category-card,
    .feature-item,
    .app-card {
        opacity: 0;
        transform: translateY(30px);
        transition: all 0.6s ease;
    }
    
    .animate-in {
        opacity: 1 !important;
        transform: translateY(0) !important;
    }
`;
document.head.appendChild(style);
```
