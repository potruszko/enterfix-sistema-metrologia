    init() {
        this.bindEvents();
        this.updateCompareButton();
    }
    
    bindEvents() {
        // Checkboxes de seleção
        document.querySelectorAll('.selector-item input').forEach(checkbox => {
            checkbox.addEventListener('change', (e) => {
                this.handleProductSelection(e);
            });
        });
        
        // Botão comparar
        document.querySelector('.compare-btn').addEventListener('click', () => {
            this.generateComparison();
        });
    }
    
    handleProductSelection(event) {
        const productId = event.target.closest('.selector-item').dataset.product;
        
        if (event.target.checked) {
            if (this.selectedProducts.length < this.maxComparison) {
                this.selectedProducts.push(productId);
            } else {
                event.target.checked = false;
                alert(`Você pode comparar no máximo ${this.maxComparison} produtos.`);
                return;
            }
        } else {
            this.selectedProducts = this.selectedProducts.filter(id => id !== productId);
        }
        
        this.updateCompareButton();
        this.updateSelectorVisual();
    }
    
    updateCompareButton() {
        const button = document.querySelector('.compare-btn');
        if (this.selectedProducts.length >= 2) {
            button.disabled = false;
            button.textContent = `Comparar ${this.selectedProducts.length} Produtos`;
        } else {
            button.disabled = true;
            button.textContent = 'Selecione ao menos 2 produtos';
        }
    }
    
    updateSelectorVisual() {
        document.querySelectorAll('.selector-item').forEach((item, index) => {
            const productId = item.dataset.product;
            if (this.selectedProducts.includes(productId)) {
                item.classList.add('selected');
                item.style.order = this.selectedProducts.indexOf(productId);
            } else {
                item.classList.remove('selected');
                item.style.order = '';
            }
        });
    }
    
    generateComparison() {
        const container = document.getElementById('comparison-results');
        const headerRow = document.getElementById('header-row');
        const tbody = document.getElementById('comparison-body');
        
        // Limpar conteúdo anterior
        headerRow.innerHTML = '<th class="feature-column">Características</th>';
        tbody.innerHTML = '';
        
        // Adicionar colunas dos produtos selecionados
        this.selectedProducts.forEach(productId => {
            const produto = this.produtos[productId];
            headerRow.innerHTML += `
                <th class="product-column">
                    <div class="product-header">
                        <h4>${produto.nome}</h4>
                        <span class="destaque">${produto.destaque}</span>
                        <a href="/${productId}" class="view-product">Ver Produto</a>
                    </div>
                </th>
            `;
        });
        
        // Características para comparação
        const caracteristicas = [
            { key: 'precisao', label: 'Precisão' },
            { key: 'taxa', label: 'Taxa de Medição' },
            { key: 'volume', label: 'Volume de Trabalho' },
            { key: 'distancia', label: 'Distância de Trabalho' },
            { key: 'peso', label: 'Peso' },
            { key: 'luz', label: 'Fonte de Luz' },
            { key: 'certificacao', label: 'Certificações' },
            { key: 'software', label: 'Software Incluído' },
            { key: 'aplicacoes', label: 'Aplicações Principais' },
            { key: 'preco', label: 'Categoria de Preço' }
        ];
        
        // Gerar linhas de comparação
        caracteristicas.forEach(caracteristica => {
            let row = `<tr class="comparison-row">
                <td class="feature-label">${caracteristica.label}</td>`;
            
            this.selectedProducts.forEach(productId => {
                const produto = this.produtos[productId];
                const valor = produto[caracteristica.key];
                const cellClass = this.getCellClass(caracteristica.key, valor);
                
                row += `<td class="comparison-cell ${cellClass}">${valor}</td>`;
            });
            
            row += '</tr>';
            tbody.innerHTML += row;
        });
        
        // Mostrar tabela
        container.style.display = 'block';
        container.scrollIntoView({ behavior: 'smooth', block: 'start' });
        
        // Adicionar linha de ações
        let actionsRow = '<tr class="actions-row"><td class="feature-label">Ações</td>';
        this.selectedProducts.forEach(productId => {
            actionsRow += `
                <td class="comparison-cell">
                    <div class="product-actions">
                        <a href="/${productId}" class="btn-detalhes">Ver Detalhes</a>
                        <a href="/contato?produto=${productId}" class="btn-demo">Agende Demonstração</a>
                    </div>
                </td>
            `;
        });
        actionsRow += '</tr>';
        tbody.innerHTML += actionsRow;
    }
    
    getCellClass(key, valor) {
        // Destacar melhores valores
        if (key === 'precisao') {
            if (valor.includes('0,015') || valor.includes('0,02')) return 'highlight-best';
            if (valor.includes('0,04') || valor.includes('0,06')) return 'highlight-good';
            return 'highlight-basic';
        }
        
        if (key === 'taxa') {
            const rate = parseFloat(valor);
            if (rate >= 3.0) return 'highlight-best';
            if (rate >= 2.0) return 'highlight-good';
            return 'highlight-basic';
        }
        
        if (key === 'peso') {
            const weight = parseFloat(valor);
            if (weight <= 0.7) return 'highlight-best';
            if (weight <= 1.0) return 'highlight-good';
            return 'highlight-basic';
        }
        
        return '';
    }
}
